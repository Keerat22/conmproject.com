                }
                
                int main() {
                    float a, b;  // Integration limits
                    int n;       // Number of intervals (must be a multiple of 3)
                    float result;

                    // Check if n is a multiple of 3
                    if (n % 3 != 0) {
                        printf("Error: Number of intervals must be a multiple of 3.\n");
                        return 1;
                    }
                
                    // Calculate the area using Simpson's 3/8 rule
                    result = simpsons_3_8_rule(function, a, b, n);
                
                    // Output the result

                    return 0;
                }
                
                // Function to calculate the integral using Simpson's 3/8 rule
                float simpsons_3_8_rule(float (*func)(float), float a, float b, int n) {
                    float h = (b - a) / n;  // Width of each interval
                    float area = func(a) + func(b);  // Initialize area with the first and last terms
                
                    // Sum the areas of the odd and even indexed terms
                    for (int i = 1; i < n; i++) {
                        float x = a + i * h;
                        if (i % 3 == 0) {
                            area += 2 * func(x);  // Every third term (even index)
                        } else {
                            area += 3 * func(x);  // Other terms (odd index)
                        }
                    }
                
                    area *= (3 * h / 8);  // Final area calculation
                    return area;          // Return total area
                }
