Lagrange's interpolation Method

Program for Lagrange's interpolation Method. 

#include < stdio.h>

            // Function to calculate Lagrange's interpolation
            float lagrange_interpolation(float x[], float y[], int n, float x_value) {
                float result = 0.0;
            
                for (int i = 0; i < n; i++) {
                    float term = y[i];
                    
                    for (int j = 0; j < n; j++) {
                        if (j != i) {
                            term *= (x_value - x[j]) / (x[i] - x[j]);
                        }
                    }
                    
                    result += term;
                }
            
                return result;
            }
            
            int main() {
                int n;
                float x[10], y[10], x_value;
            
                // Input the number of data points
                printf("Enter the number of data points: ");
                scanf("%d", &n);
            
                // Input the data points
                printf("Enter the data points (x and y values):\n");
                for (int i = 0; i < n; i++) {
                    printf("x[%d]: ", i);
                    scanf("%f", &x[i]);
                    printf("y[%d]: ", i);
                    scanf("%f", &y[i]);
                }
            
                // Input the value to interpolate
                printf("Enter the value to interpolate: ");
                scanf("%f", &x_value);
            
                // Calculate the interpolated value
                float result = lagrange_interpolation(x, y, n, x_value);
            
                // Output the result
                printf("The interpolated value at x = %.2f is: %.2f\n", x_value, result);
            
                return 0;
            }
