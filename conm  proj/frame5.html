Gauss Elimination Method

Program for Gauss Elimination Method. 

            #include < stdio.h>
                #include < stdlib.h>
                
                #define MAX 10  // Maximum number of variables
                
                // Function to perform the Gauss Elimination
                void gauss_elimination(float a[MAX][MAX], float b[MAX], int n) {
                    // Forward elimination
                    for (int i = 0; i < n; i++) {
                        // Partial pivoting
                        for (int k = i + 1; k < n; k++) {
                            if (a[k][i] > a[i][i]) {
                                for (int j = 0; j < n; j++) {
                                    float temp = a[i][j];
                                    a[i][j] = a[k][j];
                                    a[k][j] = temp;
                                }
                                float temp = b[i];
                                b[i] = b[k];
                                b[k] = temp;
                            }
                        }
                
                        // Elimination process
                        for (int k = i + 1; k < n; k++) {
                            float factor = a[k][i] / a[i][i];
                            for (int j = i; j < n; j++) {
                                a[k][j] -= factor * a[i][j];
                            }
                            b[k] -= factor * b[i];
                        }
                    }
                
                    // Back substitution
                    float x[MAX];
                    for (int i = n - 1; i >= 0; i--) {
                        x[i] = b[i];
                        for (int j = i + 1; j < n; j++) {
                            x[i] -= a[i][j] * x[j];
                        }
                        x[i] /= a[i][i];
                    }
                
                    // Print the solution
                    printf("The solution is:\n");
                    for (int i = 0; i < n; i++) {
                        printf("x[%d] = %.2f\n", i, x[i]);
                    }
                }
                
                int main() {
                    int n;
                    float a[MAX][MAX], b[MAX];
                
                    // Input the number of variables
                    printf("Enter the number of variables: ");
                    scanf("%d", &n);
                
                    // Input the coefficients of the equations
                    printf("Enter the coefficients of the equations (Augmented Matrix):\n");
                    for (int i = 0; i < n; i++) {
                        for (int j = 0; j < n; j++) {
                            scanf("%f", &a[i][j]);
                        }
                        scanf("%f", &b[i]);
                    }
                
                    // Perform Gauss elimination
                    gauss_elimination(a, b, n);
                
                    return 0;
                }
