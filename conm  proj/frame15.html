Simpsons 1/3 rule Method

Program for Simpsons 1/3 rule Method. 

            #include < stdio.h>

                // Function prototype
                float simpsons_rule(float (*func)(float), float a, float b, int n);
                
                // Example function to integrate
                float function(float x) {
                    return x * x;  // Example: f(x) = x^2
                }
                
                int main() {
                    float a, b;  // Integration limits
                    int n;       // Number of intervals (must be even)
                    float result;
                
                    // Input the limits of integration
                    printf("Enter the lower limit (a): ");
                    scanf("%f", &a);
                    printf("Enter the upper limit (b): ");
                    scanf("%f", &b);
                    printf("Enter the number of intervals (n, must be even): ");
                    scanf("%d", &n);
                
                    // Check if n is even
                    if (n % 2 != 0) {
                        printf("Error: Number of intervals must be even.\n");
                        return 1;
                    }
                
                    // Calculate the area using Simpson's 1/3 rule
                    result = simpsons_rule(function, a, b, n);
                
                    // Output the result
                    printf("The approximate integral from %.2f to %.2f is: %.2f\n", a, b, result);
                
                    return 0;
                }
                
                // Function to calculate the integral using Simpson's 1/3 rule
                float simpsons_rule(float (*func)(float), float a, float b, int n) {
                    float h = (b - a) / n;  // Width of each interval
                    float area = func(a) + func(b);  // Initialize area with the first and last terms
                
                    // Sum the areas of the odd and even indexed terms
                    for (int i = 1; i < n; i++) {
                        float x = a + i * h;
                        if (i % 2 == 0) {
                            area += 2 * func(x);  // Even indexed terms
                        } else {
                            area += 4 * func(x);  // Odd indexed terms
                        }
                    }
                
                    area *= (h / 3);  // Final area calculation
                    return area;      // Return total area
                }
