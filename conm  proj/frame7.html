Guass jordan Method

Program for Guass jordan Method. 

#include < stdio.h>
            #include < stdlib.h>
            
            #define MAX 10  // Maximum number of variables
            
            // Function to perform Gauss-Jordan elimination
            void gauss_jordan(float a[MAX][MAX], int n) {
                for (int i = 0; i < n; i++) {
                    // Make the diagonal contain all 1s
                    float pivot = a[i][i];
                    for (int j = 0; j <= n; j++) {
                        a[i][j] /= pivot;
                    }
            
                    // Make the other rows contain 0s in the current column
                    for (int k = 0; k < n; k++) {
                        if (k != i) {
                            float factor = a[k][i];
                            for (int j = 0; j <= n; j++) {
                                a[k][j] -= factor * a[i][j];
                            }
                        }
                    }
                }
            
                // Print the solution
                printf("The solution is:\n");
                for (int i = 0; i < n; i++) {
                    printf("x[%d] = %.2f\n", i, a[i][n]);
                }
            }
            
            int main() {
                int n;
                float a[MAX][MAX];
            
                // Input the number of variables
                printf("Enter the number of variables: ");
                scanf("%d", &n);
            
                // Input the coefficients of the equations (Augmented Matrix)
                printf("Enter the coefficients of the equations (Augmented Matrix):\n");
                for (int i = 0; i < n; i++) {
                    for (int j = 0; j <= n; j++) {
                        scanf("%f", &a[i][j]);
                    }
                }
            
                // Perform Gauss-Jordan elimination
                gauss_jordan(a, n);
            
                return 0;
            }
